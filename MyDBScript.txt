CREATE TABLE [dbo].[tblUser](
[uId]	INT		IDENTITY (1, 1) NOT NULL PRIMARY KEY,
[Username] VARCHAR (50) NOT NULL,
[Password] VARCHAR (50) NOT NULL,
[FullName] VARCHAR (50) NULL,
[UserType] VARCHAR (50) NULL

);
GO

CREATE TABLE [dbo].[tblCustomer] (
[cId]	INT	IDENTITY (1, 1) NOT NULL PRIMARY KEY,
[Name] VARCHAR (50) NOT NULL,
[Email] VARCHAR (50) NOT NULL,
[Phone] VARCHAR (50) NOT NULL,
[uId]	INT	NULL,
FOREIGN KEY ([uId]) REFERENCES [dbo].[tblUser]([uId]) 
ON DELETE CASCADE
);
GO

CREATE TABLE [dbo].[tblMonth](
[Id] INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
[MonthName] VARCHAR (50) NOT NULL,
[uId] INT NOT NULL,
FOREIGN KEY ([uId]) REFERENCES [dbo].[tblUser]([uId])
ON DELETE CASCADE
);
GO

CREATE TABLE [dbo].[tblExpenseCategory](
[ecId] INT IDENTITY (1,1) NOT NULL PRIMARY KEY,
[CategoryName] VARCHAR (50) NOT NULL,
[UserId] INT NOT NULL,
FOREIGN KEY ([UserId]) REFERENCES [dbo].[tblUser]([uId])
ON DELETE CASCADE
); 
GO

CREATE TABLE [dbo].[tblExpenseItem](
[eiId]	INT	IDENTITY (1,1)	NOT NULL PRIMARY KEY,
[ItemName]	VARCHAR (50)	NOT NULL,
[Cost]	MONEY	NOT NULL,
[Description]	VARCHAR (50) NOT NULL,
[ecId]	INT NOT NULL,
[uId] INT NOT NULL,
FOREIGN KEY ([uId]) REFERENCES [dbo].[tblUser]([uId]),
FOREIGN KEY ([ecId]) REFERENCES [dbo].[tblExpenseCategory] ([ecId])
ON DELETE CASCADE
);
GO

CREATE TABLE [dbo].[tblPaymentType]
(
	[ID] INT IDENTITY (1,1) NOT NULL 	PRIMARY KEY , 
    [PaymentTypeName] VARCHAR(50) NOT NULL,
	[UserID] INT NOT NULL,
	FOREIGN KEY ([UserID]) REFERENCES [dbo].[tblUser]([uID])
	ON DELETE CASCADE

);
GO

CREATE TABLE [dbo].[tblAccount](
[aId]	INT	IDENTITY (1,1)	NOT NULL PRIMARY KEY,
[cId]	INT NULL,
[openDate]	DATE NULL,
[closeDate]	DATE NULL,
[status]	BIT NULL,
FOREIGN KEY ([cId]) REFERENCES [dbo].[tblCustomer] ([cId])
ON DELETE CASCADE
);
GO
CREATE TABLE [dbo].[tblExpenseRecord] (
    [erId]        INT          IDENTITY (1, 1) NOT NULL,
	[MonthId]	  INT		    NOT NULL,
    [PaymentTypeId] INT		   NOT NULL,
    [Cost]        DECIMAL (18)       NOT NULL,
    [ecId]        INT          NULL,
    [Description]       VARCHAR (500) NULL,
    [expitemId]         INT          NULL,
    [UserId]      INT          NOT NULL,
	
    PRIMARY KEY CLUSTERED ([erId] ASC),
    FOREIGN KEY ([UserId]) REFERENCES [dbo].[tblUser] ([uId]),
    FOREIGN KEY ([ecId]) REFERENCES [dbo].[tblExpenseCategory] ([ecId]),
    FOREIGN KEY ([MonthId]) REFERENCES [dbo].[tblMonth] ([Id]),
	FOREIGN KEY ([PaymentTypeId]) REFERENCES [dbo].[tblPaymentType] ([Id]),
	FOREIGN KEY ([expitemId]) REFERENCES [dbo].[tblExpenseItem] ([eiId])
	ON DELETE CASCADE

);

GO
